    //Functions can be passed as arguments to other functions:
    function applyFn(fn, arg1, arg2) {
        return fn(arg1, arg2);
    }
    function add(x, y) { return x + y; }
    function multiply(x, y) { return x * y; }
    var a = applyFn(add, 4, 5); // 9
    var b = applyFn(multiply, 4, 5); // 20
