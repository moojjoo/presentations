Person.prototype = Object.create(null, {
        firstName: { value: "John", writable: true },
        lastName: {
            value: "Doe",
            writable: true,
            configurable: false,
            enumerable: false
        },
        birthday: { value: function() {
            return "Happy Birthday";
        }},
        _address: { value: null, writable: true },
        address: {
            get: function () {
                return this._address;
            },
            set: function(value) {
                this._address = value;
            }
        }
    });
